348   private Bitmap getResizedBitmap(Bitmap image, int maxSize) {
349     int width = image.getWidth();
350     int height = image.getHeight();
351 
352     float bitmapRatio = (float) width / (float) height;
353     if (bitmapRatio > 1) {
354       width = maxSize;
355       height = (int) (width / bitmapRatio);
356     } else {
357       height = maxSize;
358       width = (int) (height * bitmapRatio);
359     }
360 
361     Bitmap resizedBitmap = Bitmap.createScaledBitmap(image, width, height, true);
362     ByteArrayOutputStream out = new ByteArrayOutputStream();
363     // quality is ignored for PNGs
364     resizedBitmap.compress(Bitmap.CompressFormat.JPEG, QUALITY, out);
365     byte[] compressedBitmap = out.toByteArray();
366 
367     return BitmapFactory.decodeByteArray(compressedBitmap, 0, compressedBitmap.length);
368   }
