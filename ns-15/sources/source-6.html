173   private Integer passwordFormIsValid() {
174     if (password.getValue() == null || password.getValue().equals("")) {
175       return R.string.no_password_error;
176     }
177     if (password.getValue().length() < 8) {
178       return R.string.password_to_short_error;

181     if (!m.matches()) {

183     }
184     return null;
185   }
